// ----------------------------- // Helpers // ----------------------------- function clamp(v, min, max) { return Math.max(min, Math.min(max, v)); } function lerp(a, b, t) { return a + (b - a) * t; } function dist2(x1, y1, x2, y2) { const dx = x2 - x1;
